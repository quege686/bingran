// Search filter
function filterTable() {
  var query = document.getElementById('searchBox').value.toLowerCase().trim();
  var sections = document.querySelectorAll('.dim-section');
  var noResults = document.getElementById('noResults');
  var visibleCount = 0;
  
  sections.forEach(function(section) {
    var rows = section.querySelectorAll('tbody tr');
    var hasVisible = false;
    
    rows.forEach(function(row) {
      var searchText = row.getAttribute('data-search') || '';
      if (!query || searchText.indexOf(query) !== -1) {
        row.style.display = '';
        hasVisible = true;
        visibleCount++;
      } else {
        row.style.display = 'none';
      }
    });
    
    section.style.display = hasVisible ? '' : 'none';
  });
  
  noResults.style.display = visibleCount === 0 ? 'block' : 'none';
  
  // Update stats
  var stats = document.getElementById('stats');
  if (query) {
    stats.textContent = '筛选结果：' + visibleCount + ' 条匹配';
  } else {
    stats.textContent = stats.getAttribute('data-original') || stats.textContent;
  }
}

// Save original stats
document.getElementById('stats').setAttribute('data-original', document.getElementById('stats').textContent);

// Copy to clipboard
function copyText(btn, text) {
  // Use clipboard API
  if (navigator.clipboard) {
    navigator.clipboard.writeText(text).then(function() {
      btn.classList.add('copied');
      btn.textContent = '✓ 已复制';
      setTimeout(function() {
        btn.classList.remove('copied');
        btn.textContent = '复制';
      }, 1500);
    });
  } else {
    // Fallback
    var textarea = document.createElement('textarea');
    textarea.value = text;
    document.body.appendChild(textarea);
    textarea.select();
    try {
      document.execCommand('copy');
      btn.classList.add('copied');
      btn.textContent = '✓ 已复制';
      setTimeout(function() {
        btn.classList.remove('copied');
        btn.textContent = '复制';
      }, 1500);
    } catch(e) {}
    document.body.removeChild(textarea);
  }
}

    btn.classList.add('show');
  } else {
    btn.classList.remove('show');
  }
});
